{

 n1+=n;

 }

for(i=0;i<n;i++)

rt[i]=bt[i];
}

 totalt+=bt[i];
}

 if(at[i]>at[j])
 {

void studentqueries()
{
 printf("\n -----------Students Qureies ------------------ \n");
 j=0;
 printf("\nQuery Number\tSolved Time\n");
 while(j<=n)
 {
 j++;
 for(i=0;i<n;i++)
 {
 if(rt[i]==0) continue;
 if(rt[i]>ts)
 {
 printf("SQ%d\t\t%d\n",qname[i],q);
 q=q+ts;
 rt[i]=rt[i]-ts;
 te[i]=te[i]+1;
 }
 else
 {
 printf("SQ%d\t\t%d\n",qname[i],q);
 wt[i]=q-te[i]*ts;
 q=q+rt[i];
 rt[i]=rt[i]-rt[i];
 }
 }
 }
}
